#!/bin/bash
projects_json=$(curl -X GET "<REGISTRY URL:í¬íŠ¸>/v2/_catalog")
projects=($(echo $projects_json | jq -r '.repositories[]'))

for project in "${projects[@]}"; do

  echo "project name: $project"
  tags_json=$(curl -s -X GET "<REGISTRY URL:í¬íŠ¸>/v2/$project/tags/list")
  tags=($(echo $tags_json | jq -r '.tags[] | select(. != "latest")'))

  for tag in "${tags[@]}"; do

    echo "tag: $tag"
    digest=$(curl -v --silent -H "Accept: application/vnd.docker.distribution.manifest.v2+json" \
      -X GET "<REGISTRY URL:í¬íŠ¸>/v2/$project/manifests/$tag" 2>&1 | grep docker-content-digest | awk '{print $3}')

    if [ ! -z "$digest" ]; then
      echo "digest: $digest"
      digest=$(echo $digest | tr -d '\r')
      delete_response=$(curl -v --silent -H "Accept: application/vnd.docker.distribution.manifest.v2+json" \
        -X DELETE "<REGISTRY URL:í¬íŠ¸>/v2/$project/manifests/$digest" 2>&1)
      echo "Delete response: $delete_response"
    else
      echo "Digest not found - tag: $tag"
    fi
  done

  exec=$(docker exec -it registry registry garbage-collect /etc/docker/registry/config.yml)
  list=$(curl -X GET <REGISTRY URL:í¬íŠ¸>/v2/$project/tags/list)
  echo "Result: $list"

done
